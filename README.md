Func(Int) FizzBuzz(n) {
	Int cnt = 1
	while cnt<=n {
		if cnt%15 == 0 {
			Print("FizzBuzz")
		}
		elseif cnt%3 == 0 {
			Print("Fizz")
		}
		elseif cnt%5 == 0 {
			Print("Buzz")
		}
		else {
			Print(cnt.ToString)
		}
	}
}
Func(String[]):Int main(args) {
	FizzBuzz(100)
	return 0
}
